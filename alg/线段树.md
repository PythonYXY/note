# 732 我的日程安排表

//当 k 个日程安排有一些时间上的交叉时（例如 k 个日程安排都在同一时间内），就会产生 k 次预订。 
//
// 给你一些日程安排 [start, end) ，请你在每个日程安排添加后，返回一个整数 k ，表示所有先前日程安排会产生的最大 k 次预订。 
//
// 实现一个 MyCalendarThree 类来存放你的日程安排，你可以一直添加新的日程安排。 
//
// 
// MyCalendarThree() 初始化对象。 
// int book(int start, int end) 返回一个整数 k ，表示日历中存在的 k 次预订的最大值。 
// 
//
// 
//
// 示例： 
//
// 
//输入：
//["MyCalendarThree", "book", "book", "book", "book", "book", "book"]
//[[], [10, 20], [50, 60], [10, 40], [5, 15], [5, 10], [25, 55]]
//输出：
//[null, 1, 1, 2, 3, 3, 3]
//
//解释：
//MyCalendarThree myCalendarThree = new MyCalendarThree();
//myCalendarThree.book(10, 20); // 返回 1 ，第一个日程安排可以预订并且不存在相交，所以最大 k 次预订是 1 次预订。
//myCalendarThree.book(50, 60); // 返回 1 ，第二个日程安排可以预订并且不存在相交，所以最大 k 次预订是 1 次预订。
//myCalendarThree.book(10, 40); // 返回 2 ，第三个日程安排 [10, 40) 与第一个日程安排相交，所以最大 k 次预订是
// 2 次预订。
//myCalendarThree.book(5, 15); // 返回 3 ，剩下的日程安排的最大 k 次预订是 3 次预订。
//myCalendarThree.book(5, 10); // 返回 3
//myCalendarThree.book(25, 55); // 返回 3
// 
//
// 
//
// 提示： 
//
// 
// 0 <= start < end <= 10⁹ 
// 每个测试用例，调用 book 函数最多不超过 400次 
// 
// Related Topics 设计 线段树 二分查找 有序集合 👍 170 👎 0


//leetcode submit region begin(Prohibit modification and deletion)
class MyCalendarThree {

    class Node {
        int left, right, add, max;
    }

    Node[] tree = new Node[50000];

    int cnt = 0, N = (int)1e9;

    public MyCalendarThree() {
        tree[0] = new Node();
    }

    private void update(int cur, int lc, int rc, int l, int r, int val) {
        if (l <= lc && r >= rc) {
            tree[cur].add += val;
            tree[cur].max += val;
            return;
        }
        lazyCreate(cur);
        int mid = (lc + rc) >> 1;
        if (l <= mid) update(tree[cur].left, lc, mid, l, r, val);
        if (r > mid) update(tree[cur].right, mid + 1, rc, l, r, val);

        // push up
        tree[cur].max = Math.max(tree[tree[cur].left].max, tree[tree[cur].right].max);
    }

    private int query(int cur, int lc, int rc, int l, int r) {
        if (l <= lc && r >= rc) {
            return tree[cur].max;
        }
        lazyCreate(cur);
        int mid = (lc + rc) >> 1;
        int ans = 0;
        if (l <= mid) ans = query(tree[cur].left, lc, mid, l, r);
        if (r > mid) ans = Math.max(ans, query(tree[cur].right, mid + 1, rc, l , r));
        return ans;
    }

    private void lazyCreate(int cur) {
        if (tree[cur] == null) {
            tree[cur] = new Node();
        }
        if (tree[cur].left == 0) {
            tree[cur].left = ++cnt;
            tree[tree[cur].left] = new Node();
        }
        if (tree[cur].right == 0) {
            tree[cur].right = ++cnt;
            tree[tree[cur].right] = new Node();
        }

        // push down
        tree[tree[cur].left].add += tree[cur].add;
        tree[tree[cur].left].max += tree[cur].add;
        tree[tree[cur].right].add += tree[cur].add;
        tree[tree[cur].right].max += tree[cur].add;

        tree[cur].add = 0;
    }


    public int book(int start, int end) {
        update(0, 0, N, start, end - 1, 1);
        return query(0, 0, N, 0, N);
    }
}
