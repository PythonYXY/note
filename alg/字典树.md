class MagicDictionary {
    class Node {
        boolean isEnd;
        Node[] next;
    }

    Node[] dict = new Node[26];


    public MagicDictionary() {

    }

    public void buildDict(String[] dictionary) {
        for (String word: dictionary) buildDict(dict, word, 0);
    }

    private void buildDict(Node[] dict, String word, int idx) {
        if (idx == word.length()) return;
        char c = word.charAt(idx);
        int charIdx = c - 'a';
        if (dict[charIdx] == null) {
            Node node = new Node();
            node.next = new Node[26];
            dict[charIdx] = node;
        }
        if (idx == (word.length() - 1)) {
            dict[charIdx].isEnd = true;
            return;
        }
        buildDict(dict[charIdx].next, word, idx + 1);
    }

    public boolean search(String searchWord) {
        return query(dict, searchWord, 0, 1, false);
    }

    private boolean query(Node[] dict, String searchWord, int idx, int limit, boolean isEnd) {
        if (limit < 0) return false;
        if (idx == searchWord.length()) return isEnd && limit == 0;
        char c = searchWord.charAt(idx);
        for (int i = 0; i < 26; i++) {
            if (dict[i] == null) continue;
            if (query(dict[i].next, searchWord, idx + 1,
                    c - 'a' == i ? limit : limit - 1, dict[i].isEnd)) return true;
        }
        return false;
    }
}
